# {frontmatter.title}

{frontmatter.description}

```ts
type Callback<S> = (state: S) => void | (() => void | undefined);

export const useStateWithCallback = <S>(initialState: S, callback: Callback<S>) => {
  const [state, setState] = useState(initialState);

  const didMount = useRef(false);

  useEffect(() => {
    if (didMount.current) {
      callback(state);
    } else {
      didMount.current = true;
    }
  }, [state, callback]);

  return [state, setState];
};
```

## Usage

```tsx
const App = () => {
  // plain toggle
  const [isOn, toggleIsOn] = useToggle();
  // toggle with callback
  const [isOnWithCallback, toggleIsOnWithCallback] = useToggle(false, () =>
    console.log("they called back"),
  );
  return (
    <>
      {isOn ? "The light is on!" : "Hey who turned off the lights"}
      <button onClick={toggleIsOn}>Press me</button>
      <button onClick={() => toggleIsOn(() => console.log("individual callback"))}>Press me</button>
      <button onClick={() => toggleIsOnWithCallback(() => console.log("individual callback"))}>
        Press me
      </button>
    </>
  );
};
```
