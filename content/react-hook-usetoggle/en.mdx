# {frontmatter.title}

{frontmatter.description}

```ts
type noop = () => void;

export function useToggle(
  initialValue = false,
  callback?: () => void,
): [boolean, (x?: noop | any) => void] {
  const callbackRef = useRef<noop | null>(null);
  const [value, setValue] = useState(initialValue);

  useEffect(() => {
    if (value && typeof callback === "function") callback();

    if (typeof callbackRef.current === "function") {
      callbackRef.current();

      callbackRef.current = null;
    }
  }, [value]);

  const toggle = useCallback((callback): void => {
    setValue((v) => !v);
    callbackRef.current = callback;
  }, []);

  return [value, toggle];
}
```

## Usage

```tsx
const App = () => {
  // plain toggle
  const [isOn, toggleIsOn] = useToggle();
  // toggle with callback
  const [isOnWithCallback, toggleIsOnWithCallback] = useToggle(false, () =>
    console.log("they called back"),
  );
  return (
    <>
      {isOn ? "The light is on!" : "Hey who turned off the lights"}
      <button onClick={toggleIsOn}>Press me</button>
      <button onClick={() => toggleIsOn(() => console.log("individual callback"))}>Press me</button>
      <button onClick={() => toggleIsOnWithCallback(() => console.log("individual callback"))}>
        Press me
      </button>
    </>
  );
};
```
